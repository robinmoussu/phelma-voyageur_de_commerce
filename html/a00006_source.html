46 void read_arcs(FILE *fp, Arc arcs[], Ville villes[], int nb_villes, int nb_arcs)
47 {
48  int noeud1, noeud2; // noeud1 et noeud2 les deux ville d'une arrete

50 
51  // lecture des Arcs
52  flush_line(fp); // lire les mots"Arêtes du graphe : noeud1 noeud2 valeur"
53  for (i = 0; i < nb_arcs; i++) { // lire toutes les info dans "sommets du graphe"
54  Arc *p_arc = get_in_arcs(arcs, i);
55  Ville *p_ville;
56 
57  fscanf(fp,"%d %d %lf", &noeud1, &noeud2, &(p_arc->distance));
58  p_arc->pheromonesAB = p_arc->pheromonesBA = EPSILON_PHEROMONES;
59 
60  // on cherche la ville de départ et d'arrivée à partir de leurs identifiants
61  // et on ajoute l'arc courant dans la liste des voisins
62  // NB : les arcs sont supposé bi-directionnels
63  p_ville = get_in_villes(villes,noeud1,nb_villes); // ville de départ
64  p_ville->voisins[(p_ville->nb_voisins)++] = p_arc;
65  p_arc->ville_A = (struct Ville*) p_ville;
66  p_ville = get_in_villes(villes,noeud2,nb_villes); // ville d'arrivée
67  p_ville->voisins[(p_ville->nb_voisins)++] = p_arc;
68  p_arc->ville_B = (struct Ville*) p_ville;
69  }
