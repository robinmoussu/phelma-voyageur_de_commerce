72 void* memory_allocator(Ville *(villes[]), Arc *(arcs[]), Fourmi *(*fourmis[]), Fourmi **meilleure_fourmi,
73  bool *(ville_visitees[]), double *(proba_ville[]), int nb_villes, int nb_arcs, int nb_fourmis)
74 {
75  void **memory_pool;
76  Fourmi *fourmis_pool;
77  int i;
78 
79 
80  memory_pool = malloc (
81  nb_villes*sizeof_one_ville(nb_villes)
82  + nb_arcs*sizeof_one_arc(nb_arcs)
83  + nb_fourmis*sizeof(Fourmi*)
84  + nb_fourmis*sizeof_one_fourmi(nb_villes)
85  + sizeof_one_fourmi(nb_villes)
86  + sizeof_villes_visitee(nb_villes)
87  + sizeof_proba_ville(nb_arcs)
88  );
89  // Sous linux c'est inutile vu que malloc renvoie toujours un pointeur valide
90  if (memory_pool==NULL) {
91  fprintf(stderr, "Error : Allocation error\n");
92  return 0;
93  }
94 
95  // On fait correspondre chaque pointeur à son espace mémoire
96  *villes = (Ville*) memory_pool;
97  *arcs = (Arc*) ( (void*) *villes + nb_villes*sizeof_one_ville(nb_villes));
98  *fourmis = (Fourmi**) ( (void*) *arcs + nb_arcs*sizeof_one_arc(nb_arcs));
99  fourmis_pool = (Fourmi*) ( (void*) *fourmis + nb_fourmis*sizeof(Fourmi*));
100  *meilleure_fourmi = (Fourmi*) ( (void*) fourmis_pool + nb_fourmis*sizeof_one_fourmi(nb_villes));
101  *ville_visitees = (bool*) ( (void*) *meilleure_fourmi + sizeof_one_fourmi(nb_villes));
102  *proba_ville = (double*) ( (void*) *ville_visitees + sizeof_villes_visitee(nb_villes));
103 
104  // On initialise les pointeurs sur les fourmis
105  for (i = 0; i < NB_FOURMIS; ++i) {
106  (*fourmis)[i] = (Fourmi*) ((void*) fourmis_pool + i*sizeof_one_fourmi(nb_villes));
107  }
108 
109  return memory_pool;
110 }
