51 void ville_suivante(Fourmi *f, int alpha, int beta, double proba_ville[], bool deja_visite[])
52 {
53  double tirage;
54  double cumul_proba;
55  double fx;
56  Ville *ville_courante;
57  int i;
58 
59  ville_courante = f->tabu[f->nb_villes_deja_visite - 1];
60 
61  // On calule la probabilitée de tirer chacune des ville à visiter
62  cumul_proba = 0;
63  for (i = 0; i < ville_courante->nb_voisins; i++) {
64  Arc *arc_courant = ville_courante->voisins[i];
65 
66  if (deja_visite[get_arrivee(ville_courante, arc_courant)->id_ville]) {
67  proba_ville[i] = 0;
68  } else {
69  proba_ville[i] = pow(*get_pheromones(ville_courante,arc_courant),alpha) / pow(arc_courant->distance,beta);
70 
71  }
72 
73  cumul_proba += proba_ville[i];
74  }
75 
76  if (cumul_proba == 0) {
77  ON_VERBOSE(fprintf(stderr,"error: All cities were already visited, no path found (it's normal in incomplete graph)\n"));
78  f->parcourt_valide = false;
79  return;
80  }
81 
82  // On selectionne aléatoirement une ville parmi la liste de ville (en la normalisant)
83  tirage = rand() * cumul_proba / (double) RAND_MAX;
84  fx = 0;
85  for (i = 0; i < ville_courante->nb_voisins; i++) {
86  fx += proba_ville[i];
87  if ((tirage < fx) && (proba_ville[i] != 0)) {
88  // C'est cette ville qui a été selectionnée
89  Ville *ville_arrivee = get_arrivee(ville_courante, ville_courante->voisins[i]);
90 
91  f->tabu[f->nb_villes_deja_visite++] = ville_arrivee;
92  f->L += get_arc(ville_courante, ville_arrivee)->distance; // On actualise la distance parcouru
93  deja_visite[ville_arrivee->id_ville] = true;
94  ON_DEBUG(printf("City chosen : %s\n", ville_arrivee->nom);)
95  break;
96  }
97  }

117 bool parcourt_valide(Fourmi *f, int nb_villes, bool ville_visitees[])
118 {
119  int i;
120  double distance = 0;
121 
122  // on s'assure que la fourmi semble avoir visiter toutes les villes
123  if (f->nb_villes_deja_visite != nb_villes) {
124  fprintf(stderr, "Error : All cities are not visited (%d cities visited instead of %d)\n",f->nb_villes_deja_visite, nb_villes);
125  return false;
126  }
127 
128  // On initialise la liste des villes visitées
129  memset(ville_visitees, false, nb_villes*sizeof(bool));
130 
131  // On refait le parcourt de la fourmi (de la ville 0 à l'avant-dernière ville vu qu'on calcule la distance entre la ville n et n+1)
132  ville_visitees[f->tabu[0]->id_ville] = true;
133  for (i = 0; i < (nb_villes - 1); i++) {
134  Ville *ville_courante = f->tabu[i];
135  Ville *ville_suivante = f->tabu[i + 1];
136 
137  // On s'assure que la fourmi n'était pas déjà passé par cette ville
138 
139  if (ville_visitees[ville_suivante->id_ville]) {
140  fprintf(stderr, "Error : City visited twice (%s)\n", ville_suivante->nom);
141  return false;
142  } else {
143  ville_visitees[ville_suivante->id_ville] = true;
144  }
145 
146  // on calcule la distance jusqu'à la prochaine ville
147  distance += get_arc(ville_courante, ville_suivante)->distance;
148  }
149 
150  // on vérifie que la distance était correctement calculée
151  // vu que ce sont des flotants, on ne peut pas faire (distance == f->L)
152  if ((distance - f->L) > 0.00001) {
153  fprintf(stderr, "error : Distance invalid\n");
154  return false;
155  }
156 
157  return true;
158 }

160 void parcourt_update(Fourmi **fourmi_actuelle, Fourmi **meilleure_fourmi, int nb_villes, bool ville_visitees[], double evaporation, double depot_pheromones)
161 {
162  Ville *depart, *arrivee;
163  int i;
164 
165  // on s'assure que le parcourt de la fourmi est valide
166  ON_DEBUG(
167  if (!parcourt_valide(*fourmi_actuelle, nb_villes, ville_visitees)) {
168  fprintf(stderr, "Error : Invalid path.\n");
169  return;
170  }
171  )
172 
173  depart = (*fourmi_actuelle)->tabu[0];
174  for(i = 1; i < nb_villes; i++) { // nb : on commence avec la ville numéro 1 (soit l'arc de la ville 0 vers la ville 1)
175  Arc *arc_courant;
176  arrivee = (*fourmi_actuelle)->tabu[i];
177  arc_courant= get_arc(depart, arrivee);
178 
179  // on met à jour les phéromones sur les arcs
180  *get_pheromones(depart,arc_courant) = evaporation*(arc_courant->pheromonesAB) + depot_pheromones/((*fourmi_actuelle)->L);
181 
182  // on actualise la ville de départ
183  depart = arrivee;
184  }
185 
186  // On regarde si la nouvelle fourmi a un trajet plus performant que l'ancienne (et que l'ancien était initialisé)
187  // Si c'est le cas, (plutot que de copier toutes les données de la fourmi actuelle dans la meilleure fourmi,
188  // ce qui est long), on les swap.
189  if ((*fourmi_actuelle)->L < (*meilleure_fourmi)->L) {
190  swap((void*) fourmi_actuelle, (void*) meilleure_fourmi);
191  }
192 }
