67 void* memory_allocator(Ville *(villes[]), Arc *(arcs[]), Fourmi **meilleure_fourmi, Fourmi **fourmi_actuelle,
68  bool *(ville_visitees[]), float *(proba_ville[]), int nb_ville, int nb_arc)
69 {
70  void *memory_pool;
71 
72  memory_pool = malloc (
73  nb_ville*sizeof_one_ville(nb_ville)
74  + nb_arc*sizeof_one_arc(nb_arc)
75  + 2*sizeof_fourmi(nb_ville)
76  + sizeof_villes_visitee(nb_ville)
77  + sizeof_proba_ville(nb_arc)
78  );
79  // Sous linux c'est inutile vu que malloc renvoie toujours un pointeur valide
80  if (memory_pool==NULL) {
81  fprintf(stderr, "Error : Allocation error\n");
82  return 0;
83  }
84 
85  // On fait correspondre chaque pointeur à son espace mémoire
86  *villes = (Ville*) memory_pool;
87  *arcs = (Arc*) ( (void*)*villes + nb_ville*sizeof_one_ville(nb_ville));
88  *fourmi_actuelle = (Fourmi*) ( (void*)*arcs + nb_arc*sizeof_one_arc(nb_arc));
89  *meilleure_fourmi = (Fourmi*) ( (void*)*fourmi_actuelle + sizeof_fourmi(nb_ville));
90  *ville_visitees = (bool*) ( (void*)*meilleure_fourmi + sizeof_fourmi(nb_ville));
91  *proba_ville = (float*) ( (void*)*ville_visitees + sizeof_villes_visitee(nb_ville));
92 
93  return memory_pool;
94 }
