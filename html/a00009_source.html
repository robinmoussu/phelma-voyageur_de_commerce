47 void ville_suivante(Fourmi *f, int nb_villes, int alpha, int beta, float proba_ville[], bool deja_visite[])
48 {
49  float tirage;
50  float cumul_proba;
51  float fx;
52  Ville *ville_courante;
53  int i;
54 
55  ville_courante = f->tabu[f->nb_villes_deja_visite - 1];
56 
57  // On calule la probabilitée de tirer chacune des ville à visiter
58  cumul_proba = 0;
59  for (i = 0; i < ville_courante->nb_voisins; i++) {
60  Arc *arc_courant = ville_courante->voisins[i];
61 
62  if (deja_visite[get_arrivee(ville_courante, arc_courant)->id_ville]) {
63  proba_ville[i] = 0;
64  } else {
65  proba_ville[i] = pow(arc_courant->pheromones,alpha) / pow(arc_courant->distance,beta);
66  }
67 
68  cumul_proba += proba_ville[i];
69  }
70 
71  if (cumul_proba == 0) {
72  fprintf(stderr,"error: All cities were already visited, no path found\n");
73  return;
74  }
75 
76  // On selectionne aléatoirement une ville parmi la liste de ville (en la normalisant)
77  tirage = rand() * cumul_proba / (float) RAND_MAX;
78  fx = 0;
79  for (i = 0; i < ville_courante->nb_voisins; i++) {
80  fx += proba_ville[i];
81  if ((tirage < fx) && (proba_ville[i] != 0)) {
82  // C'est cette ville qui a été selectionnée
83  Ville *ville_arrivee = get_arrivee(ville_courante, ville_courante->voisins[i]);
84 
85  f->tabu[f->nb_villes_deja_visite++] = ville_arrivee;
86  f->L += get_arc(ville_courante, ville_arrivee)->distance; // On actualise la distance parcouru
87  deja_visite[ville_arrivee->id_ville] = true;
88  ON_DEBUG(printf("City chosen : %s\n", ville_arrivee->nom);)
89  break;
90  }
91  }
92 
93 }

111 bool parcourt_valide(Fourmi *f, Ville villes[], int nb_villes, bool ville_visitees[])
112 {
113  int i, j;
114  double distance = 0;
115 
116  // on s'assure que la fourmi semble avoir visiter toutes les villes
117  if (f->nb_villes_deja_visite != nb_villes) {
118  fprintf(stderr, "Error : All cities are not visited (%d cities visited instead of %d)\n",f->nb_villes_deja_visite, nb_villes);
119  return false;
120  }
121 
122  // On initialise la liste des villes visitées
123  ville_visitees[0] = true;
124  memset(ville_visitees + 1, false, nb_villes*sizeof(bool));
125 
126  // On refait le parcourt de la fourmi (de la ville 0 à l'avant-dernière ville vu qu'on calcule la distance entre la ville n et n+1)
127  for (i = 0; i < (nb_villes - 1); i++) {
128  Ville *ville_courante = f->tabu[i];
129  Ville *ville_suivante = f->tabu[i + 1];
130 
131  // On s'assure que la fourmi n'était pas déjà passé par cette ville
132 
133  if (ville_visitees[ville_suivante->id_ville]) {
134  fprintf(stderr, "Error : City visited twice (%s)\n", ville_suivante->nom);
135  return false;
136  } else {
137  ville_visitees[ville_suivante->id_ville] = true;
138  }
139 
140  // on calcule la distance jusqu'à la prochaine ville
141  distance += get_arc(ville_courante, ville_suivante)->distance;
142  }
143 
144  // on vérifie que la distance était correctement calculée
145  // vu que ce sont des flotants, on ne peut pas faire (distance == f->L)
146  if ((distance - f->L) > 0.00001) {
147  fprintf(stderr, "error : Distance invalid\n");
148  return false;
149  }
150 
151  return true;
152 }

154 void graph_update(Fourmi **fourmi_actuelle, Fourmi **meilleure_fourmi, Ville villes[], int nb_villes, bool ville_visitees[])
155 {
156  Ville *depart, *arrivee;
157  int i;
158 
159  // on s'assure que le parcourt de la fourmi est valide
160  if (!parcourt_valide(*fourmi_actuelle, villes, nb_villes, ville_visitees)) {
161  fprintf(stderr, "Error : Invalid path.\n");
162  return;
163  }
164 
165  depart = (*fourmi_actuelle)->tabu[0];
166  for(i = 1; i < nb_villes; i++) { // nb : on commence avec la ville numéro 1 (soit l'arc de la ville 0 vers la ville 1)
167  arrivee = (*fourmi_actuelle)->tabu[i];
168 
169  // on met à jour les phéromones sur les arcs
170  get_arc(depart, arrivee)->pheromones += PARCOURT_PHEROMONES;
171 
172  // on actualise la ville de départ
173  depart = arrivee;
174  }
175 
176  // On regarde si la nouvelle fourmi a un trajet plus performant que l'ancienne (et que l'ancien était initialisé)
177  // Si c'est le cas, (plutot que de copier toutes les données de la fourmi actuelle dans la meilleure fourmi,
178  // ce qui est long), on les swap.
179  if ((*fourmi_actuelle)->L < (*meilleure_fourmi)->L) {
180  swap(fourmi_actuelle,meilleure_fourmi);
181  }
182 }
